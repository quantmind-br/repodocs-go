func ValidateLLMProvider(s string) error {
    if s == "" {
        return nil
    }
    validProviders := map[string]bool{
        "openai":    true,
        "anthropic": true,
        "google":    true,
    }
    if !validProviders[strings.ToLower(s)] {
        return fmt.Errorf("invalid LLM provider: must be openai, anthropic, or google")
    }
    return nil
}
